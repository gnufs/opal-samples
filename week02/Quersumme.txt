digitSum: N x N -> N
digitSum inputNumber

setze decimal <- 1
setze result <- 0

SOLANGE (inputNumber > (decimal * 10))
   TUE 
      (
      setze decimal <- (decimal + 1)
      )

SOLANGE (decimal > 0)
   TUE (
      setze decimalBase <- 1
      setze i <- decimal   
      SOLANGE (i > 1)
         TUE (
            setze decimalBase <- (decimalBase * 10)
            setze i <- (i - 1)
             )
      setze result <- (result + ((inputNumber - mod(inputNumber,decimalBase)) / decimal))
      setze decimal <- (decimal - 1)
       )
   
Ergebnis is result
